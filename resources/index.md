<section class="section-card">
  <h2>App Terms</h2>
  <div class="grid cols-2">
    <div class="card accent-gold">
      <h3>Disney Premier Access</h3>
      <p>Paid access for eligible attractions or viewing areas with less waiting.</p>
    </div>
    <div class="card accent-land">
      <h3>Priority Pass</h3>
      <p>Free access for eligible attractions after park entry, when available.</p>
    </div>
    <div class="card accent-sea">
      <h3>Standby Pass</h3>
      <p>Sometimes required for select rides, shops, or areas depending on the day.</p>
    </div>
    <div class="card accent-gold">
      <h3>Entry Request</h3>
      <p>Used for certain shows and greetings through the app.</p>
    </div>
    <div class="card accent-land">
      <h3>Mobile Order</h3>
      <p>Lets you reserve food pickup times so lunch does not eat your afternoon.</p>
    </div>
    <div class="card accent-sea">
      <h3>Priority Seating</h3>
      <p>Advance reservations for select restaurants.</p>
    </div>
  </div>
</section>
